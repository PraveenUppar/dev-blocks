// ============ CONTROLLER ============
export async function getPublishedPostController(
  req: Request,
  res: Response,
  next: NextFunction,
) {
  try {
    // Validate and sanitize inputs
    const page = Math.max(1, parseInt(req.query.page as string) || 1);
    const rawLimit = parseInt(req.query.limit as string) || 10;
    const limit = Math.min(50, Math.max(1, rawLimit)); // Cap at 50
    
    // Optional: Add sorting and filtering
    const sortBy = (req.query.sortBy as string) || 'latest';
    const tag = req.query.tag as string;

    const posts = await getPublishedPostService({ 
      page, 
      limit, 
      sortBy, 
      tag 
    });
    
    return res.status(200).json({ 
      success: true, 
      ...posts 
    });
  } catch (error) {
    next(error);
  }
}

// ============ SERVICE ============
import NodeCache from 'node-cache';
const cache = new NodeCache({ stdTTL: 300 }); // 5 minutes

interface GetPostsParams {
  page: number;
  limit: number;
  sortBy?: string;
  tag?: string;
}

export async function getPublishedPostService({
  page,
  limit,
  sortBy = 'latest',
  tag,
}: GetPostsParams) {
  // Cache key includes all params
  const cacheKey = `posts:${page}:${limit}:${sortBy}:${tag || 'all'}`;
  const cached = cache.get(cacheKey);
  if (cached) return cached;

  const skip = (page - 1) * limit;

  // Build where clause
  const where: any = {
    status: "PUBLISHED",
  };
  
  // Filter by tag if provided
  if (tag) {
    where.tags = {
      some: {
        tag: {
          slug: tag,
        },
      },
    };
  }

  // Determine sort order
  const orderBy: any = {};
  switch (sortBy) {
    case 'latest':
      orderBy.publishedAt = 'desc';
      break;
    case 'oldest':
      orderBy.publishedAt = 'asc';
      break;
    case 'popular':
      orderBy.viewCount = 'desc';
      break;
    case 'trending':
      // Combine likes + recency (you'd need a more complex query)
      orderBy.likes = { _count: 'desc' };
      break;
    default:
      orderBy.publishedAt = 'desc';
  }

  const [posts, total] = await Promise.all([
    prisma.post.findMany({
      where,
      skip,
      take: limit,
      orderBy,
      select: {
        id: true,
        title: true,
        subtitle: true,
        slug: true,
        coverImage: true,
        readTime: true,
        viewCount: true,
        publishedAt: true,
        createdAt: true,
        author: {
          select: {
            id: true,
            username: true,
            name: true,
            avatar: true,
          },
        },
        tags: {
          select: {
            tag: {
              select: { 
                name: true, 
                slug: true 
              },
            },
          },
        },
        _count: {
          select: { 
            comments: true, 
            likes: true 
          },
        },
      },
    }),
    prisma.post.count({ where }),
  ]);

  const result = {
    data: posts.map(post => ({
      ...post,
      tags: post.tags.map(pt => pt.tag), // Flatten tags
    })),
    pagination: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
      hasNextPage: page < Math.ceil(total / limit),
      hasPrevPage: page > 1,
    },
  };

  cache.set(cacheKey, result);
  return result;
}
